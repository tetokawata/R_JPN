Chapter 7 条件付き平均効果関数の推定

library(tidyverse)

data("NMES1988",
     package = "AER")

raw <- na.omit(NMES1988)

raw <- mutate(raw,
              no_insurance = if_else(insurance == "no",1, 0))


Y <- raw$visits

D <- raw$no_insurance

X <- model.matrix(~ - 1+ region + age + afam + gender + school,
                  raw)

set.seed(123)

7.1 Casual Forest

library(grf)

fit <- regression_forest(X = X,
                         Y = Y)

Y.hat <- predict(fit)$predictions

fit <- regression_forest(X = X,
                         Y = D)

D.hat <- predict(fit)$predictions

fit.CF <- causal_forest(X = X,
                        W = D,
                        Y = Y,
                        Y.hat = Y.hat,
                        W.hat = D.hat)

df <- mutate(raw,
             tau.grf = predict(fit.CF)$predictions)

7.2 Distribution of predicted individual effects

ggplot(df,
       aes(x = tau.grf)
       ) +
  geom_histogram()
